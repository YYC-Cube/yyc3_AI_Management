## 4. 集成测试

### API 接口测试

```typescript
// backend/src/__tests__/integration/tickets.integration.test.ts
import request from 'supertest';
import app from '../../server';

describe('Tickets API - Integration Tests', () => {
  let authToken: string;

  beforeAll(async () => {
    // 获取认证token
    const res = await request(app)
      .post('/api/auth/login')
      .send({ email: 'test@example.com', password: 'password123' });
    authToken = res.body.accessToken;
  });

  describe('GET /api/tickets', () => {
    it('should return tickets list with authentication', async () => {
      const response = await request(app)
        .get('/api/tickets')
        .set('Authorization', `Bearer ${authToken}`)
        .expect(200);

      expect(Array.isArray(response.body)).toBe(true);
    });

    it('should reject unauthenticated requests', async () => {
      await request(app)
        .get('/api/tickets')
        .expect(401);
    });
  });
});
```

## 7. CI/CD 集成

### GitHub Actions 工作流

```yaml
# .github/workflows/test-gate.yml
name: Test Gate

on:
  pull_request:
    branches: [main, develop]

jobs:
  test:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        node-version: [18.x, 20.x]

    steps:
      - uses: actions/checkout@v3
      
      - name: Setup Node.js ${{ matrix.node-version }}
        uses: actions/setup-node@v3
        with:
          node-version: ${{ matrix.node-version }}
      
      - name: Install dependencies
        run: npm ci
      
      - name: Run unit tests
        run: npm test -- --coverage
      
      - name: Run integration tests
        run: npm run test:integration
      
      - name: Upload coverage report
        uses: codecov/codecov-action@v3
        with:
          files: ./coverage/lcov.info
```

## 10. 最佳实践

### ✅ 推荐做法

1. **遵循 AAA 模式**
   - Arrange（准备）: 设置测试数据和 mock
   - Act（执行）: 调用被测函数
   - Assert（断言）: 验证结果

2. **测试命名规范**
   ```
   should_[expected_behavior]_when_[condition]
   
   示例：
   - should_return_ticket_list_when_authenticated
   - should_throw_error_for_invalid_input
   ```

3. **保持测试独立性**
   - 每个测试应该独立运行
   - 不依赖执行顺序
   - 清理测试数据

4. **使用有意义的断言**
   ```typescript
   // ❌ Bad
   expect(result).toBeTruthy();
   
   // ✅ Good
   expect(result.id).toBeDefined();
   expect(result.title).toBe(expectedTitle);
   expect(result.createdAt).toBeInstanceOf(Date);
   ```

### ❌ 常见错误

1. **测试实现细节而非行为**
2. **过度使用 mock 导致测试脆弱**
3. **忽略边界情况和异常处理**
4. **测试间共享状态导致不确定性**

#### Step 2: 编写测试模板

```typescript
// backend/src/__tests__/helpers/test-helpers.ts
import { Request, Response, NextFunction } from 'express';

export const mockRequest = (overrides = {}) => ({
  headers: {},
  body: {},
  params: {},
  query: {},
  user: { id: 'test-user-id', email: 'test@example.com' },
  ...overrides,
}) as Partial<Request>;

export const mockResponse = () => {
  const res = {} as Response;
  res.status = jest.fn().mockReturnValue(res);
  res.json = jest.fn().mockReturnValue(res);
  res.send = jest.fn().mockReturnValue(res);
  return res;
};

export const mockNext = () => jest.fn() as NextFunction;
```

#### Step 3: 批量生成测试骨架

```bash
# 为所有 service 文件生成测试骨架
find backend/src/services -name "*.ts" ! -name "*.test.ts" ! -name "*.spec.ts" | while read file; do
  test_file="${file%.ts}.test.ts"
  if [ ! -f "$test_file" ]; then
    cat > "$test_file" << EOF
import { $(basename ${file%.*}) } from '../$(basename $file)';

describe($(basename ${file%.*}), () => {
  // TODO: Add test cases
});
EOF
    echo "Created: $test_file"
  fi
done
```
